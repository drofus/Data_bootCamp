üßÆ Score RICE#

import pandas as pd

rice_df = pd.DataFrame({
    "idea": ["Idea A", "Idea B"],
    "alcance": [1000, 5000],
    "impacto": [3, 2],
    "esfuerzo": [20, 10],
    "confianza": [0.9, 0.7]
})

rice_df["score"] = (rice_df["alcance"] * rice_df["impacto"] * rice_df["confianza"]) / rice_df["esfuerzo"]
rice_df.sort_values("score", ascending=False)
